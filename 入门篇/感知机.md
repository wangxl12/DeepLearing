"""感知机的实现原理是一个分段函数，
如果输入的x1 * w1 + x2 * w2 + b > theta ， 
则输出1，否则输出0."""
import numpy as np
import matplotlib.pyplot as plt


def AND(x1, x2):   # 根据感知机原理实现‘与’门,则此时的theta=0
    x = np.array([x1, x2])
    w = np.array([0.2, 0.3])  # 通过改变权重值可以实现‘或’门，# ‘非’门
    b = 0.1
    k = np.sum(w * x) + b
    return k


"""感知机通过改变w和b的值来实现与或非门，那么什么样的值合适呢"""
x = np.linspace(0, 10, 20)
y = -1 * x + 1
plt.plot(x, y)
plt.show()
"""通过图像可见直线和x, y交点是（1，0），（0，1）。
这样在直线上方的输出1，在下方的输出0，可以实现‘与’门。
只有同为1的时候才输出1，其他情况下输出0。同理可以实现或非、门。
但是实现不了与非门，与非门：相同为0，不同为1，在图像画出来不是线性函数，要实现与非门这样的非线性关系还需要组合门"""
